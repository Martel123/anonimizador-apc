def ensure_superadmin_user():
    from models import User, db
    from werkzeug.security import generate_password_hash
    import os, logging

    email = (os.environ.get("SUPERADMIN_EMAIL") or "").strip().lower()
    password = os.environ.get("SUPERADMIN_PASSWORD") or ""
    if not email or not password:
        logging.info("SUPERADMIN env vars not set; skipping superadmin seed")
        return

    user = User.query.filter_by(email=email).first()
    if user:
        changed = False
        if getattr(user, "role", None) != "superadmin":
            user.role = "superadmin"
            changed = True
        if getattr(user, "activo", None) is not True:
            user.activo = True
            changed = True
        if changed:
            db.session.commit()
            logging.info("SUPERADMIN updated (role/activo)")
        else:
            logging.info("SUPERADMIN exists; no changes")
        return

    admin = User(
        email=email,
        role="superadmin",
        activo=True,
        password_hash=generate_password_hash(password),
    )
    db.session.add(admin)
    db.session.commit()
    logging.info("SUPERADMIN created")
